CREATE TABLE job (ID int  NOT NULL AUTO_INCREMENT ,name varchar(128),jenkins_url varchar(128),job_type int,status int,features mediumtext,description mediumtext , gmt_create datetime DEFAULT NULL,gmt_modified datetime DEFAULT NULL,PRIMARY KEY (id))ENGINE=InnoDB DEFAULT CHARSET=utf8;
CREATE TABLE jobs_relation (ID int  NOT NULL AUTO_INCREMENT ,pipeline_id int not null,job_id int not null,up_stream_jobs varchar(255),down_stream_jobs varchar(255),features mediumtext,description mediumtext , gmt_create datetime DEFAULT NULL,gmt_modified datetime DEFAULT NULL,PRIMARY KEY (id))ENGINE=InnoDB DEFAULT CHARSET=utf8;
CREATE TABLE pipeline (ID int  NOT NULL AUTO_INCREMENT ,name varchar(128),group_id int,head_job_id int,features mediumtext,description mediumtext , gmt_create datetime DEFAULT NULL,gmt_modified datetime DEFAULT NULL,PRIMARY KEY (id))ENGINE=InnoDB DEFAULT CHARSET=utf8;
CREATE TABLE pipeline_group (ID int  NOT NULL AUTO_INCREMENT ,name varchar(128),features mediumtext,description mediumtext , gmt_create datetime DEFAULT NULL,gmt_modified datetime DEFAULT NULL,PRIMARY KEY (id))ENGINE=InnoDB DEFAULT CHARSET=utf8;
